<script>
function pickCar() {
    const budget = document.getElementById("budget").value;
    const type = document.getElementById("type").value;
    let car = "";

    if (budget === "low" && type === "city") car = "Volkswagen Golf";
    else if (budget === "mid" && type === "family") car = "Toyota Camry";
    else if (budget === "high" && type === "sport") car = "BMW M4";
    else car = "Skoda Octavia";

    document.getElementById("result").innerText = "Рекомендуем: " + car;
}
</script>
